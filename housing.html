function yearsBetween(start, end){

  if(!start||!end) return 0;

  const s=new Date(start), e=new Date(end);

  if(e<s) return 0;

  let y=e.getFullYear()-s.getFullYear();

  if(e.getMonth()<s.getMonth()||(e.getMonth()===s.getMonth()&&e.getDate()<s.getDate())) y--;

  return y<0?0:y;

}


function calculate(){

  const birth=document.getElementById('birth').value;

  const married=document.getElementById('married').value;

  const custom=document.getElementById('customStart').value;

  const join=document.getElementById('joinDate').value;

  const deps=parseInt(document.getElementById('dependents').value||'0',10);

  const todayStr=new Date().toISOString().slice(0,10);


  let mujuStart="";

  if(custom) mujuStart=custom;

  else if(married) mujuStart=married;

  else if(birth){

    const b=new Date(birth);

    const thirtieth=new Date(b.getFullYear()+30,b.getMonth(),b.getDate());

    mujuStart=thirtieth.toISOString().slice(0,10);

  }


  const mujuYears=yearsBetween(mujuStart,todayStr);

  const mujuPoints=Math.min(32,mujuYears*2);

  const famPoints=Math.min(35,deps*5);

  const joinYears=yearsBetween(join,todayStr);

  const joinPoints=(joinYears>=15)?17:joinYears;

  const total=mujuPoints+famPoints+joinPoints;


  document.getElementById('output').style.display='block';

  document.getElementById('totalP').textContent='총점: '+total+'점';

  document.getElementById('detail').textContent=

    `무주택 ${mujuYears}년(${mujuPoints}점) · 부양가족 ${deps}명(${famPoints}점) · 가입 ${joinYears}년(${joinPoints}점)`;

  document.getElementById('mujuY').textContent=mujuYears+'년';

  document.getElementById('mujuP').textContent=mujuPoints+'점';

  document.getElementById('famCount').textContent=deps+'명';

  document.getElementById('famP').textContent=famPoints+'점';

  document.getElementById('joinY').textContent=joinYears+'년';

  document.getElementById('joinP').textContent=joinPoints+'점';

}
